x3 = float(input("x3 мәнін енгізіңіз: "))

slope2 = (y4 - y3) / (x4 - x3) if x4 != x3 else float('inf')

    print("Түзулер параллель")

        y = slope1 * (x - x1) + y1

        print("Қиылысу жоқ немесе ерекше жағдай")
